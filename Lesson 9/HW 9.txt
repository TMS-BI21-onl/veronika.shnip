SELECT p1.FirstName + ' ' + p1.LastName AS Dmitriev_FIO,
       p2.FirstName + ' ' + p2.LastName AS Father_FIO
FROM People p1
JOIN People p2 ON p1.ID_Father = p2.ID /*p1.ID_Father = p2.ID связывает записи, где отец имеет ID, соответствующий ID_Father в записи сына.
                                         Условие p1.LastName = 'Дмитриев' фильтрует только записи с фамилией Дмитриев.*/
WHERE p1.LastName = 'Дмитриев'
